    Quickstart

    1. Install
    $ npm install --save kinesis-exchange/grpc-methods
    1. Require
    const { GrpcUnaryMethod } = require('grpc-methods')
    1. Enjoy
    const server = new grpc.Server()

server.addService(serviceDefinition, {
  myMethod: new GrpcUnaryMethod(myMethod).register()
})
